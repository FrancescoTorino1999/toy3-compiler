// Funzione per dividere un numero per 10 senza usare il modulo
def divideByTen(num : double ; ref remainder : double) : double {
    divisor : double;
    result : 0;
    remainder := num;
    divisor := 1.0;

    while (divisor <= num) do {
        divisor := divisor * 10;
    }
    divisor := divisor / 10;

    while (divisor > 0) do {
        if (remainder >= divisor) then {
            remainder := remainder - divisor;
            result := result + 1;
        } else {
            divisor := divisor / 10;
            result := result * 10;
        }
    }

    return result / 10;
}
